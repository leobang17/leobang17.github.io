class Solution:
    def jump(self, nums: List[int]) -> int:
        ptr, count = 0, 0
        
        while ptr < len(nums) - 1:
            maxJump, next = 0, 0
            #    jump    .
            if ptr + nums[ptr] >= len(nums) - 1:
                return count + 1


            for i in range(ptr + 1, ptr + nums[ptr] + 1):
                if i + nums[i] >= maxJump: # jump           .
                    maxJump = i + nums[i]
                    next = i
            ptr = next
            count += 1


        return count
